#include<stdio.h>
#include<stdlib.h>
#include<unistd.h>
#include<sys/wait.h>

int main() {
    pid_t pid;

    printf("Parent PID: %d\n", getpid());

    pid = fork();

    if(pid < 0) {
        printf("Fork failed\n");
        exit(1);
    }

    if(pid == 0) {
        // Child process
        printf("Child Process\n");
        printf("Child PID: %d\n", getpid());
        printf("Parent PID: %d\n", getppid());
        
        sleep(2);
        printf("Child exiting\n");
        exit(0);
    } 
    else {
        // Parent process
        wait(NULL);
        printf("Parent resumed\n");
        printf("Parent exiting\n");
    }

    return 0;
}

9.THREADING

#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>

sem_t mutex, wrt;
int readcount = 0;

// Reader Thread
void* reader(void* arg) {
    sem_wait(&mutex);
    readcount++;
    if (readcount == 1)
        sem_wait(&wrt);   // first reader blocks writer
    sem_post(&mutex);

    printf("Reader is reading\n");
    sleep(1);

    sem_wait(&mutex);
    readcount--;
    if (readcount == 0)
        sem_post(&wrt);   // last reader releases writer
    sem_post(&mutex);

    return NULL;
}

// Writer Thread
void* writer(void* arg) {
    sem_wait(&wrt);

    printf("Writer is writing\n");
    sleep(1);

    sem_post(&wrt);
    return NULL;
}

int main() {
    pthread_t r1, r2, w1;

    sem_init(&mutex, 0, 1);
    sem_init(&wrt, 0, 1);

    // Create threads
    pthread_create(&r1, NULL, reader, NULL);
    pthread_create(&r2, NULL, reader, NULL);
    pthread_create(&w1, NULL, writer, NULL);

    // Wait for threads
    pthread_join(r1, NULL);
    pthread_join(r2, NULL);
    pthread_join(w1, NULL);

    return 0;
}

#include <stdio.h>

int main() {

    int n, r, i, j;

    int alloc[10][10], req[10][10];

    int avail[10], finish[10]={0};

    printf("Enter no of processes: ");
    scanf("%d",&n);

    printf("Enter no of resources: ");
    scanf("%d",&r);

    printf("Enter Allocation Matrix:\n");
    for(i=0;i<n;i++)
        for(j=0;j<r;j++)
            scanf("%d",&alloc[i][j]);

    printf("Enter Request Matrix:\n");
    for(i=0;i<n;i++)
        for(j=0;j<r;j++)
            scanf("%d",&req[i][j]);

    printf("Enter Available Resources:\n");
    for(i=0;i<r;i++)
        scanf("%d",&avail[i]);

    int count=0;

    while(count<n){

        int found=0;

        for(i=0;i<n;i++){

            if(finish[i]==0){

                int possible=1;

                for(j=0;j<r;j++){

                    if(req[i][j] > avail[j]){

                        possible=0;
                        break;
                    }
                }

                if(possible){

                    for(j=0;j<r;j++)
                        avail[j]+=alloc[i][j];

                    finish[i]=1;

                    count++;

                    found=1;
                }
            }
        }

        if(found==0){

            printf("Deadlock Detected");
            return 0;
        }
    }

    printf("No Deadlock");
}

#include <stdio.h>
#include<stdlib.h>
int mutex=1;
int full=0;
int empty;
int buffersize;
    
int signal(int s){
    return ++s;
}
    
int wait(int s){
    return --s;
}
    
void producer(){
    mutex=wait(mutex);
    empty=wait(empty);
    full=signal(full);
    printf("full:%d|empty:%d",full,empty);
    mutex=signal(mutex);
}
    
void consumer(){
    mutex=wait(mutex);
    full=wait(full);
    empty=signal(empty);
    printf("full:%d|empty:%d",full,empty);
    mutex=signal(mutex);    
}
int main(){
    int choice;
    printf("enter buffersize:");
    scanf("%d",&buffersize);
    empty=buffersize;
    while(1){
        printf("\n1.consumer\t 2. producer \t 3.exit\n");
        printf("\nenter the choice:");
        scanf("%d",&choice);
        switch (choice) {
            case 1:
                if (full != 0) consumer();
                else printf("Buffer Empty!\n");
                break; // Essential!

            case 2:
                if (empty != 0) producer();
                else printf("Buffer Full!\n");
                break; // Essential!

            case 3:
                exit(0);

            default:
                printf("Invalid choice\n");
        }
    }
}

/#include <stdio.h>

int main() {
    int n, wt[20], bt[20], rem_bt[20], tat[20], ft[20];
    int i, tq, remain, time = 0;
    float avg_tat = 0, avg_wt = 0;

    printf("enter number of processes:");
    scanf("%d", &n);

    remain = n;

    for(i = 0; i < n; i++) {
        printf("enter the burst time for process %d:", i+1);
        scanf("%d", &bt[i]);
        rem_bt[i] = bt[i];
        tat[i] = 0;   // FIX
    }

    printf("enter the time quantum:");
    scanf("%d", &tq);

    for(i = 0; remain != 0; ) {
        if(rem_bt[i] <= tq && rem_bt[i] > 0) {
            time += rem_bt[i];
            rem_bt[i] = 0;
            remain--;

            ft[i] = time;             
            tat[i] = ft[i];           // AT = 0
            wt[i] = tat[i] - bt[i];
        }
        else if(rem_bt[i] > 0) {
            rem_bt[i] -= tq;
            time += tq;
        }

        if(i == n - 1)
            i = 0;
        else
            i++;
    }

    printf("\nProcess\tBT\tFT\tWT\tTAT\n");
    for(i = 0; i < n; i++) {
        avg_tat += tat[i];
        avg_wt += wt[i];
        printf("P%d\t%d\t%d\t%d\t%d\n", i+1, bt[i], ft[i], wt[i], tat[i]);
    }

    printf("\nAverage Waiting Time = %.2f", avg_wt / n);
    printf("\nAverage Turnaround Time = %.2f", avg_tat / n);

    return 0;
}

#include<stdio.h>
#include<stdlib.h>
#include<unistd.h>
#include<sys/wait.h>

int main() {
    pid_t pid;

    printf("Parent PID: %d\n", getpid());

    pid = fork();

    if(pid < 0) {
        printf("Fork failed\n");
        exit(1);
    }

    if(pid == 0) {
        // Child process
        printf("Child Process\n");
        printf("Child PID: %d\n", getpid());
        printf("Parent PID: %d\n", getppid());
        
        sleep(2);
        printf("Child exiting\n");
        exit(0);
    } 
    else {
        // Parent process
        wait(NULL);
        printf("Parent resumed\n");
        printf("Parent exiting\n");
    }

    return 0;
}
